rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    match /rental-contracts/{contractId}/{fileName=**} {
      allow read: if true;
      allow write: if request.resource != null
        && request.resource.size < 5242880
        && (request.resource.contentType.matches('image/.*') || request.resource.contentType == 'application/pdf');
    }

    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
